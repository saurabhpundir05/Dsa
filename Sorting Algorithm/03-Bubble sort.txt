//Bubble sort
#include<iostream>
using namespace std;
int sorting(int a[],int n)
{
  bool s;
  for(int i=0;i<n-1;i++){
    s=false;
    for(int j=0;j<n-1;j++){
      if(a[j]>a[j+1]){
        swap(a[j],a[j+1]);
        s=true;
      }
    }
    if(s==false){
      break;
    }
  }
}
void trav(int a[],int n)
{
    for(int i=0;i<n;i++){
        cout<<a[i]<<" ";
    }
}
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    sorting(a,n);
    cout<<"after sort"<<endl;
    trav(a,n);
}
